var sfw = $("#wizard").stepFormWizard({
    showNav: 'right',
    height: 'auto',
    nextBtn: $('<a class="next-btn sf-right sf-btn" href="#">FORWARD</a>'),
    prevBtn: $('<a class="prev-btn sf-left sf-btn" href="#">BACK</a>'),
    finishBtn: $('<a class="finish-btn sf-right sf-btn" href="#">DONE</a>'),
    onFinish: function(i, wizard) {
        var form = $("#wizard").serialize();
        sfw.addSpinner('finish');
        $.getJSON('form/get-finish-url', form, function(data) {
            wizard.html(data.html);
        });
        return false;
    }
});

var hashtag = window.location.hash.match(/^#step-(\d+)/);
var startStep = 0;
if(hashtag !== null) {
    startStep = hashtag[1] - 1;
}
$("#wizard").stepFormWizard({
    showButtons: false,
    startStep: startStep,
    height: 'auto',
    showNavNumbers: false,
    onNext: function(i) {
        window.location.hash = '#step-' + (i + 2);
    },
    onPrev: function(i) {
        window.location.hash = '#step-' + (i);
    }
});
